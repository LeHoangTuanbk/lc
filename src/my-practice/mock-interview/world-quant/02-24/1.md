function minimunMeetingRoom(intervals: Interval[]) {
if(intervals.length === 0) return 0;

    intervals.sort((a, b) => a[0] - b[0]); // sort by startingTime
    let pq = MinPriorityQueue<number>();
    pq.enqueue(-1);

    let minimumRoom = 1;

    for(const [startTime, endTime] of intervals) {
        const earliestEndingTime = pq.dequeue();
        if(startTime >= earliestEndingTime) {
            pq.enqueue(endTime);
        } else {
            minimumRoom++;
            pq.enqueue(endTime);
            // Edited, add after mock interview
            pq.enqueue(earliestEndingTime);
        }
    }

    return minimumRoom;

}
